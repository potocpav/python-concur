Concur

Build interactive data analysis and visualization GUI in Python.

Concur is a library for rapidly developing user interfaces. Responsive interfaces can be written with just a few lines of code.

  • Hello, world!
  • Counter
  • Image
  • Plotting
  • Animation 
import concur as c

c.main("Hello World", c.text("Hello, world!"), 400, 288)
import concur as c

def counter():
    count = 5
    while True:
        tag, _ = yield from c.orr([
            c.text(f"Count: {count}"),
            c.button("Increment"),
            c.button("Decrement"),
            ])
        if tag == "Increment":
            count += 1
        elif tag == "Decrement":
            count -= 1
        yield

c.main("Counter", counter(), 400, 288)

import concur as c
import numpy as np

def line(tf):
    np.random.seed(0)
    x = np.linspace(-2, 2, 10000)
    y = np.cumsum(np.random.rand(len(x)) - 0.5) / 100
    return c.draw.polyline(np.stack([x, y]).T, (0, 0, 1, 1), tf=tf)

def plot():
    frame = c.Frame((-1, 1), (1, -1))
    while True:
        _, frame = yield from c.frame("Frame", frame, line)
        yield

c.main("Plot", plot(), 400, 288)
import concur as c
import numpy as np
from time import monotonic as t

def graph(tf):
    ts = np.linspace(t(), t() + 1/2, 100)
    pts = np.stack([np.sin(ts * 4), np.cos(ts * 5)]).T
    return c.draw.polyline(pts, (0,0,1,1), tf=tf)

def app():
    view = c.Frame((-1.5, 1.5), (1.5, -1.5))
    while True:
        key, value = yield from c.orr([
            c.frame("Frame", view, graph),
            c.event(("Tick", None)), # refresh every frame
            ])
        if key == "Frame":
            view = value
        yield

if __name__ == "__main__":
    c.main("Plot", app(), 400, 288)

Benefits

  • Useful for simple visualizations, as well as interactive applications.
  • Works well with Jupyter
  • Real-time performant plotting
  • Vector-based image overlays
  • Scales to complicated applications without code becoming spaghetti
  • The Elm Architecture (Model-View-Update architecture) can be recovered in a few lines of code
